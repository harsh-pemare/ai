### Experiment 2: To Implement uninformed search methods using C,python or Java.
---

graph = {
    0: [1, 2], 1: [0, 3, 4], 2: [0, 4],
    3: [1, 4, 5], 4: [1, 2, 3, 5], 5: [3, 4]
}

def dfs(node, visited=set()):
    if node not in visited:
        print(node, end=" ")
        visited.add(node)
        for neighbor in graph[node]:
            dfs(neighbor, visited)

def bfs(start):
    visited, queue = set([start]), [start]
    while queue:
        node = queue.pop(0) # Pop from front of queue
        print(node, end=" ")
        for neighbor in graph[node]:
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append(neighbor)

print("DFS:"); dfs(0)
print("\nBFS:"); bfs(0)

### Experiment 3: A*/Heuristic Search (An informed search strategy-one that uses problem-specific knowledge-can find
solutions more efficiently. A key component of these algorithms is a heuristic function h(n).
h(n) = estimated cost of the cheapest path from node n to a goal node. Admissible
/heuristic never over estimated i.e. h(n)≤ Actual cost.)
---

import heapq

def a_star(graph, h, start, goal):
    # Queue stores: (f_cost, current_node)
    queue = [(0, start)]
    g_cost = {start: 0}
    path = {start: [start]}

    while queue:
        _, curr = heapq.heappop(queue)
        
        if curr == goal: 
            return path[curr], g_cost[curr]

        for neighbor, cost in graph[curr].items():
            new_g = g_cost[curr] + cost
            
            # If neighbor is unvisited or we found a shorter path to it
            if neighbor not in g_cost or new_g < g_cost[neighbor]:
                g_cost[neighbor] = new_g
                f_cost = new_g + h[neighbor]
                heapq.heappush(queue, (f_cost, neighbor))
                path[neighbor] = path[curr] + [neighbor] # Update path history
                
    return "No path", 0

facts = [("joe", "jane"), ("harry", "carl"), ("jane", "anne"), ("carl", "ralph")]

def children(parent): 
    return [c for p, c in facts if p == parent]

def parents(child): 
    return [p for p, c in facts if c == child]

def is_grandparent(gp, gc):
    # Check if the grandchild is in the children list of any of the grandparent's children
    return any(gc in children(p) for p in children(gp))

print("Children of Carl:", children("carl"))
print("Is Joe grandparent of Anne?", is_grandparent("joe", "anne"))

### Experiment 6: To implement 8 queen puzzle
---

def is_safe(board, row, col):
    for r, c in enumerate(board):
        # Check same column OR same diagonal
        if c == col or abs(r - row) == abs(c - col): 
            return False
    return True

def solve_queens(n=8, board=[]):
    if len(board) == n:
        print("First Solution (Column indices per row):", board)
        return True # Return True to stop after finding 1st solution
    
    for col in range(n):
        if is_safe(board, len(board), col):
            # Recursively place next queen
            if solve_queens(n, board + [col]): 
                return True

solve_queens()

### Experiment 7: Implement Monkey Banana Problem using Python

from collections import deque

def solve_monkey():
    # State: (monkey_pos, on_box, box_pos, has_banana)
    start = ('door', False, 'window', False)
    queue = deque([(start, [])])
    visited = set([start])

    while queue:
        (m_pos, on_box, b_pos, has_b), path = queue.popleft()
        if has_b: return path # Goal reached!

        actions = []
        # Action 1: Grasp
        if on_box and b_pos == 'center' and not has_b:
            actions.append(((m_pos, True, b_pos, True), "Grasp"))
        
        # Action 2: Climb or Push
        if m_pos == b_pos and not on_box:
            actions.append(((m_pos, True, b_pos, has_b), "Climb"))
            for pos in ['door', 'window', 'center']:
                if pos != b_pos: actions.append(((pos, False, pos, has_b), f"Push to {pos}"))
        
        # Action 3: Walk
        if not on_box:
            for pos in ['door', 'window', 'center']:
                if pos != m_pos: actions.append(((pos, False, b_pos, has_b), f"Walk to {pos}"))

        for state, action in actions:
            if state not in visited:
                visited.add(state)
                queue.append((state, path + [action]))

for step in solve_monkey(): print(step)
